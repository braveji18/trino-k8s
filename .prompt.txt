NS=user-braveji

# 0) JMX exporter ConfigMap 먼저 배포 (install-monitor.sh에도 포함되었지만 단독으로도 가능)
kubectl apply -n $NS -f manifests/monitoring/trino-jmx-exporter-config.yaml

# 1) Trino 재배포 (javaagent + annotation 적용)
helm upgrade --install my-trino my-trino/trino \
  -n $NS -f helm/values.yaml --wait --timeout 15m

# 2) Trino pod에 annotation/initContainer/볼륨이 실제로 들어갔는지 확인
POD=$(kubectl -n $NS get pod -l app.kubernetes.io/component=coordinator -o name | head -1)
kubectl -n $NS get $POD -o jsonpath='{.metadata.annotations}' | jq
kubectl -n $NS get $POD -o jsonpath='{.spec.initContainers[*].name}'
kubectl -n $NS exec $POD -c trino-coordinator -- ls -l /opt/jmx-exporter/

# 3) /metrics 엔드포인트 직접 확인
kubectl -n $NS port-forward $POD 5556:5556 &
sleep 2
curl -s http://localhost:5556/metrics | head -40
# jvm_memory_*, trino_execution_*, trino_memory_* 등이 보여야 함
kill %1

# 4) Prometheus targets 확인
kubectl -n $NS port-forward svc/prom-prometheus-server 9090:80 &
sleep 2
curl -s http://localhost:9090/api/v1/targets | jq '.data.activeTargets[] | select(.labels.job=="kubernetes-pods-slow") | {pod:.labels.kubernetes_pod_name, health, lastError}'
# my-trino-* pod가 health=up으로 보여야 함
kill %1
